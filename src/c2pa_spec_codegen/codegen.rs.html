15/// Extracts comments from raw CDDL source and exposes them keyed by line
16/// number. The `cddl` crate's pest-based parser always sets the AST
17/// `comments_*` fields to `None`, but it does preserve line spans — we
18/// use those together with this index to attach comments to generated
19/// items as Rust doc comments.
20pub(crate) struct CommentIndex<'a> {
21  lines: Vec<&'a str>,
22  /// Trailing `; ...` text for each line, without the leading `;` / space.
23  trailing: Vec<Option<String>>,
24  /// Whether each line is only a comment (optionally indented, nothing
25  /// else before the `;`).
26  is_comment_only: Vec<bool>,
27  /// The comment text for comment-only lines.
28  comment_only_text: Vec<Option<String>>,
29}
30
31impl<'a> CommentIndex<'a> {
32  pub(crate) fn build(source: &'a str) -> Self {
33    let lines: Vec<&str> = source.lines().collect();
34    let mut trailing = Vec::with_capacity(lines.len());
35    let mut is_comment_only = Vec::with_capacity(lines.len());
36    let mut comment_only_text = Vec::with_capacity(lines.len());
37
38    for line in &lines {
39      match split_comment(line) {
40        (Some(text), true) => {
41          trailing.push(None);
42          is_comment_only.push(true);
43          comment_only_text.push(Some(text));
44        }
45        (Some(text), false) => {
46          trailing.push(Some(text));
47          is_comment_only.push(false);
48          comment_only_text.push(None);
49        }
50        (None, _) => {
51          trailing.push(None);
52          is_comment_only.push(false);
53          comment_only_text.push(None);
54        }
55      }
56    }
57
58    Self {
59      lines,
60      trailing,
61      is_comment_only,
62      comment_only_text,
63    }
64  }
65
66  /// Empty index used when source text is not available.
67  pub(crate) fn empty() -> Self {
68    Self {
69      lines: Vec::new(),
70      trailing: Vec::new(),
71      is_comment_only: Vec::new(),
72      comment_only_text: Vec::new(),
73    }
74  }
75
76  /// Comment text trailing on the given 1-indexed line, if any.
77  pub(crate) fn trailing(&self, line: usize) -> Option<&str> {
78    if line == 0 || line > self.lines.len() {
79      return None;
80    }
81    self.trailing[line - 1].as_deref()
82  }
83
84  /// Block of standalone comment lines immediately preceding the given
85  /// 1-indexed line, in source order. Stops at the first blank or
86  /// non-comment line.
87  pub(crate) fn preceding(&self, line: usize) -> Vec<&str> {
88    if line <= 1 || line > self.lines.len() + 1 {
89      return Vec::new();
90    }
91    let mut result = Vec::new();
92    let mut idx = line as isize - 2;
93    while idx >= 0 {
94      let i = idx as usize;
95      if self.is_comment_only[i] {
96        result.push(self.comment_only_text[i].as_deref().unwrap_or(""));
97      } else {
98        break;
99      }
100      idx -= 1;
101    }
102    result.reverse();
103    result
104  }
105}

107/// Split a CDDL source line at the first `;` outside a string literal.
108/// Returns `(comment_text, is_comment_only)`. `comment_text` has one
109/// leading space stripped (so `; foo` yields `"foo"`).
110fn split_comment(line: &str) -> (Option<String>, bool) {
111  let mut in_string = false;
112  let mut escape = false;
113  for (byte_idx, ch) in line.char_indices() {
114    if in_string {
115      if escape {
116        escape = false;
117      } else if ch == '\\' {
118        escape = true;
119      } else if ch == '"' {
120        in_string = false;
121      }
122      continue;
123    }
124    if ch == '"' {
125      in_string = true;
126      continue;
127    }
128    if ch == ';' {
129      let before = &line[..byte_idx];
130      let after = &line[byte_idx + 1..];
131      let text = after.strip_prefix(' ').unwrap_or(after).trim_end();
132      let is_only = before.trim().is_empty();
133      return (Some(text.to_string()), is_only);
134    }
135  }
136  (None, false)
137}

220/// Generate Rust source code for a single named rule in a parsed CDDL AST.
221///
222/// If `output_name` is provided, the generated type will use that name instead
223/// of the name derived from the CDDL rule. This allows the `#[cddl]` attribute
224/// macro to preserve the user's chosen struct name.
225pub(crate) fn generate_single_type(
226  cddl: &CDDL<'_>,
227  source: &str,
228  rule_name: &str,
229  output_name: Option<&str>,
230) -> Result<String, CodegenError> {
231  let comments = CommentIndex::build(source);
232  let type_defs = collect_type_defs(cddl, &comments)?;
233  let matching = type_defs
234    .into_iter()
235    .find(|d| match d {
236      RustTypeDef::Struct { name, .. }
237      | RustTypeDef::TypeAlias { name, .. }
238      | RustTypeDef::Enum { name, .. } => name == rule_name,
239    })
240    .ok_or_else(|| {
241      CodegenError::ParseError(format!(
242        "no rule matching '{}' found in CDDL definition",
243        rule_name
244      ))
245    })?;
246
247  let mut output = String::new();
248  match &matching {
249    RustTypeDef::Struct { name, fields, docs } => {
250      let emit_name = output_name.unwrap_or(name);
251      render_struct(&mut output, emit_name, fields, docs)?;
252    }
253    RustTypeDef::TypeAlias { name, target, docs } => {
254      let emit_name = output_name.unwrap_or(name);
255      render_type_alias(&mut output, emit_name, target, docs)?;
256    }
257    RustTypeDef::Enum {
258      name,
259      variants,
260      docs,
261    } => {
262      let emit_name = output_name.unwrap_or(name);
263      render_enum(&mut output, emit_name, variants, docs)?;
264    }
265  }
266  Ok(output)
267}

269/// Convert a PascalCase struct name to a CDDL-style kebab-case identifier.
270pub(crate) fn pascal_to_cddl_name(pascal: &str) -> String {
271  let mut result = String::with_capacity(pascal.len() + 4);
272  for (i, c) in pascal.chars().enumerate() {
273    if c.is_uppercase() {
274      if i > 0 {
275        result.push('-');
276      }
277      result.push(c.to_lowercase().next().unwrap());
278    } else {
279      result.push(c);
280    }
281  }
282  result
283}

285/// Detect by-value type cycles among generated structs / enums and box
286/// one edge per cycle so the resulting Rust types have known sizes.
287///
288/// CDDL doesn't care about cycles; Rust does. A field of type `T` (or
289/// `Option<T>`) contributes T's size to the parent, so two structs that
290/// each contain the other by value compile to types of infinite size.
291/// Wrapping one edge in `Box` breaks the size cycle without changing
292/// the wire format (Box<T> serialises the same as T under serde / CBOR).
293///
294/// **Which edges count**: bare type references at the top of a field's
295/// type string (`T` or `Option<T>` with rust_type=`T`). Heap-allocated
296/// wrappers (`Vec`, `Box`, `HashMap`, `Rc`, etc.) have fixed size and
297/// don't propagate the inner type's size, so edges through them are
298/// not part of the sizing graph.
299///
300/// **Which edge to box**: DFS visits types in alphabetical order. A
301/// back-edge (one whose target is currently-on-the-stack) closes the
302/// cycle, and that edge gets boxed. The order is deterministic so
303/// regenerations produce stable diffs even as new types appear.
304fn break_type_cycles(defs: &mut [RustTypeDef]) {
305  let names: BTreeSet<String> = defs.iter().map(|d| def_name(d).to_string()).collect();
306
307  // Build adjacency in alphabetical order so the DFS is deterministic.
308  let mut adj: BTreeMap<String, BTreeSet<String>> = BTreeMap::new();
309  for def in defs.iter() {
310    let from = def_name(def).to_string();
311    let entry = adj.entry(from).or_default();
312    match def {
313      RustTypeDef::Struct { fields, .. } => {
314        for f in fields {
315          if let Some(t) = sizing_target(&f.rust_type) {
316            if names.contains(t) {
317              entry.insert(t.to_string());
318            }
319          }
320        }
321      }
322      RustTypeDef::Enum { variants, .. } => {
323        for v in variants {
324          if let Some(inner) = &v.inner_type {
325            if let Some(t) = sizing_target(inner) {
326              if names.contains(t) {
327                entry.insert(t.to_string());
328              }
329            }
330          }
331        }
332      }
333      // Type aliases are transparent; they don't allocate. Field-level
334      // boxing in the structs that point at them handles the cycle.
335      RustTypeDef::TypeAlias { .. } => {}
336    }
337  }
338
339  // DFS with white/gray/black coloring; record back-edges.
340  #[derive(Clone, Copy, PartialEq)]
341  enum Color {
342    White,
343    Gray,
344    Black,
345  }
346  let mut color: BTreeMap<String, Color> = adj.keys().map(|n| (n.clone(), Color::White)).collect();
347  let mut back_edges: HashSet<(String, String)> = HashSet::new();
348  // Iterative DFS keeps stack depth bounded for very large schemas.
349  for start in adj.keys() {
350    if color[start] != Color::White {
351      continue;
352    }
353    let mut stack: Vec<(String, std::vec::IntoIter<String>)> =
354      vec![(start.clone(), adj[start].iter().cloned().collect::<Vec<_>>().into_iter())];
355    color.insert(start.clone(), Color::Gray);
356    while let Some((node, iter)) = stack.last_mut() {
357      if let Some(next) = iter.next() {
358        match color.get(&next).copied().unwrap_or(Color::Black) {
359          Color::White => {
360            color.insert(next.clone(), Color::Gray);
361            let next_iter = adj
362              .get(&next)
363              .cloned()
364              .unwrap_or_default()
365              .into_iter()
366              .collect::<Vec<_>>()
367              .into_iter();
368            stack.push((next, next_iter));
369          }
370          Color::Gray => {
371            // Back-edge: node -> next, where next is an ancestor on the stack.
372            back_edges.insert((node.clone(), next));
373          }
374          Color::Black => {}
375        }
376      } else {
377        let (done, _) = stack.pop().unwrap();
378        color.insert(done, Color::Black);
379      }
380    }
381  }
382
383  if back_edges.is_empty() {
384    return;
385  }
386
387  // Apply the boxes: rewrite each chosen edge's field/variant type.
388  for def in defs.iter_mut() {
389    let from = def_name(def).to_string();
390    match def {
391      RustTypeDef::Struct { fields, .. } => {
392        for f in fields {
393          if let Some(t) = sizing_target(&f.rust_type) {
394            if back_edges.contains(&(from.clone(), t.to_string())) {
395              f.rust_type = format!("Box<{}>", f.rust_type);
396            }
397          }
398        }
399      }
400      RustTypeDef::Enum { variants, .. } => {
401        for v in variants {
402          if let Some(inner) = v.inner_type.clone() {
403            if let Some(t) = sizing_target(&inner) {
404              if back_edges.contains(&(from.clone(), t.to_string())) {
405                v.inner_type = Some(format!("Box<{}>", inner));
406              }
407            }
408          }
409        }
410      }
411      RustTypeDef::TypeAlias { .. } => {}
412    }
413  }
414}

416/// Extract the type name from a rust_type string if it contributes to
417/// a sizing cycle. Returns the bare name for `T`, and `None` for
418/// `Vec<T>`, `HashMap<K, V>`, `Box<T>`, paths, references, tuples,
419/// etc. — types that don't propagate size of T into the parent.
420fn sizing_target(rust_type: &str) -> Option<&str> {
421  let trimmed = rust_type.trim();
422  if trimmed.is_empty() {
423    return None;
424  }
425  // Any generic, path, reference, or tuple is either heap-allocated
426  // (Vec/Box/HashMap/...) or otherwise doesn't fit a simple bare-name
427  // edge.
428  if trimmed.contains('<') || trimmed.contains('&') || trimmed.contains('(') || trimmed.contains("::") {
429    return None;
430  }
431  Some(trimmed)
432}

511fn collect_type_defs(
512  cddl: &CDDL<'_>,
513  comments: &CommentIndex<'_>,
514) -> Result<Vec<RustTypeDef>, CodegenError> {
515  let mut defs: Vec<RustTypeDef> = Vec::new();
516  for rule in &cddl.rules {
517    match rule {
518      Rule::Type {
519        rule: type_rule,
520        span,
521        ..
522      } => {
523        // Skip rules that have a hand-written Rust substitute; the
524        // consumer crate provides the real type.
525        if rule_substitution(type_rule.name.ident).is_some() {
526          continue;
527        }
528        let docs = preceding_docs_for(comments, span);
529        if let Some(def) = type_rule_to_rust_def(type_rule, comments, docs)? {
530          push_or_merge(&mut defs, def);
531        }
532      }
533      Rule::Group {
534        rule: group_rule,
535        span,
536        ..
537      } => {
538        if rule_substitution(group_rule.name.ident).is_some() {
539          continue;
540        }
541        let name = to_pascal_case(group_rule.name.ident);
542        let docs = preceding_docs_for(comments, span);
543        if let Some(fields) = group_entry_to_fields(&group_rule.entry, comments)? {
544          push_or_merge(
545            &mut defs,
546            RustTypeDef::Struct {
547              name,
548              fields,
549              docs,
550            },
551          );
552        }
553      }
554    }
555  }
556  for def in &mut defs {
557    if let RustTypeDef::Struct { fields, .. } = def {
558      disambiguate_field_names(fields);
559    }
560  }
561  Ok(defs)
562}

573/// Push a RustTypeDef, merging into any existing definition with the same
574/// name. CDDL socket/plug extensions (`$foo /= A` then `$foo /= B`) produce
575/// multiple `Rule`s with the same identifier; emitting duplicate Rust items
576/// would not compile. This merges them by folding additional type aliases /
577/// structs / enums into an untagged enum.
578fn push_or_merge(defs: &mut Vec<RustTypeDef>, new_def: RustTypeDef) {
579  let new_name = def_name(&new_def).to_string();
580  if let Some(idx) = defs.iter().position(|d| def_name(d) == new_name) {
581    let existing = defs.remove(idx);
582    defs.insert(idx, merge_defs(existing, new_def));
583  } else {
584    defs.push(new_def);
585  }
586}

596fn merge_defs(existing: RustTypeDef, new: RustTypeDef) -> RustTypeDef {
597  let name = def_name(&existing).to_string();
598  let existing_docs = def_docs(&existing);
599  let new_docs = def_docs(&new);
600
601  let mut variants = Vec::new();
602  extend_variants_from(&mut variants, existing);
603  extend_variants_from(&mut variants, new);
604
605  let mut seen: Vec<(String, Option<String>)> = Vec::new();
606  variants.retain(|v| {
607    let key = (v.name.clone(), v.inner_type.clone());
608    if seen.contains(&key) {
609      false
610    } else {
611      seen.push(key);
612      true
613    }
614  });
615
616  let mut name_counts: std::collections::HashMap<String, usize> =
617    std::collections::HashMap::new();
618  for v in &mut variants {
619    let count = name_counts.entry(v.name.clone()).or_insert(0);
620    if *count > 0 {
621      v.name = format!("{}{}", v.name, *count + 1);
622    }
623    *count += 1;
624  }
625
626  // Keep the earliest preceding-comment block as the enum-level docs.
627  let docs = if !existing_docs.is_empty() {
628    existing_docs
629  } else {
630    new_docs
631  };
632  RustTypeDef::Enum {
633    name,
634    variants,
635    docs,
636  }
637}

647fn extend_variants_from(variants: &mut Vec<RustEnumVariant>, def: RustTypeDef) {
648  match def {
649    RustTypeDef::TypeAlias { target, docs, .. } => {
650      let variant_name = variant_name_from_type(&target);
651      variants.push(RustEnumVariant {
652        name: variant_name,
653        inner_type: Some(target),
654        serde_rename: None,
655        docs,
656      });
657    }
658    RustTypeDef::Enum {
659      variants: mut inner,
660      docs,
661      ..
662    } => {
663      // If the enum has rule-level docs and exactly one variant, carry them
664      // to that variant so merged `/=` extensions keep their line comments.
665      if !docs.is_empty() && inner.len() == 1 && inner[0].docs.is_empty() {
666        inner[0].docs = docs;
667      }
668      variants.append(&mut inner);
669    }
670    RustTypeDef::Struct { name, docs, .. } => {
671      variants.push(RustEnumVariant {
672        name: name.clone(),
673        inner_type: Some(name),
674        serde_rename: None,
675        docs,
676      });
677    }
678  }
679}

681fn variant_name_from_type(target: &str) -> String {
682  let base = target.rsplit("::").next().unwrap_or(target);
683  let base = base.split('<').next().unwrap_or(base);
684  if base.is_empty() {
685    "Variant".to_string()
686  } else {
687    to_pascal_case(base)
688  }
689}

691fn disambiguate_field_names(fields: &mut [RustField]) {
692  let mut counts: std::collections::HashMap<String, usize> =
693    std::collections::HashMap::new();
694  for f in fields.iter_mut() {
695    let entry = counts.entry(f.name.clone()).or_insert(0);
696    if *entry > 0 {
697      f.name = format!("{}_{}", f.name, *entry + 1);
698      // After disambiguation there is no single "original" key to honor;
699      // align original_name with the new Rust name so the serde rename
700      // attribute is not emitted. Multiple fields claiming the same
701      // rename produce unreachable match arms in serde's deserializer.
702      f.original_name = f.name.clone();
703    }
704    *entry += 1;
705  }
706}

708fn type_rule_to_rust_def(
709  rule: &TypeRule<'_>,
710  comments: &CommentIndex<'_>,
711  docs: Vec<String>,
712) -> Result<Option<RustTypeDef>, CodegenError> {
713  let name = to_pascal_case(rule.name.ident);
714  let ty = &rule.value;
715
716  if ty.type_choices.len() > 1 {
717    return Ok(Some(type_choices_to_enum(&name, ty, docs)?));
718  }
719
720  if let Some(tc) = ty.type_choices.first() {
721    let type1 = &tc.type1;
722    // Range expression: `a..b` / `a...b`. Alias to the primitive of the
723    // left-hand value so socket/plug merging can combine ranges with other
724    // type aliases under a single enum.
725    if type1.operator.is_some() {
726      let target = range_primitive(&type1.type2);
727      return Ok(Some(RustTypeDef::TypeAlias {
728        name,
729        target,
730        docs,
731      }));
732    }
733    match &type1.type2 {
734      Type2::Map { group, .. } => {
735        let fields = group_to_fields(group, comments)?;
736        Ok(Some(RustTypeDef::Struct { name, fields, docs }))
737      }
738      Type2::Array { group, .. } => {
739        let rust_type = array_group_to_type(group)?;
740        Ok(Some(RustTypeDef::TypeAlias {
741          name,
742          target: rust_type,
743          docs,
744        }))
745      }
746      Type2::Typename { ident, .. } => {
747        let target = cddl_ident_to_rust_type(ident.ident);
748        Ok(Some(RustTypeDef::TypeAlias { name, target, docs }))
749      }
750      Type2::ParenthesizedType { pt, .. } => {
751        if pt.type_choices.len() > 1 {
752          return Ok(Some(type_choices_to_enum(&name, pt, docs)?));
753        }
754        let target = type_to_rust_string(pt)?;
755        Ok(Some(RustTypeDef::TypeAlias { name, target, docs }))
756      }
757      Type2::Unwrap { ident, .. } => {
758        let target = to_pascal_case(ident.ident);
759        Ok(Some(RustTypeDef::TypeAlias { name, target, docs }))
760      }
761      // Single literal-value rules become a single-variant enum so that
762      // repeated `$foo /= "literal"` socket extensions can be folded into
763      // one enum by `push_or_merge`. The rule-level docs attach to the
764      // variant, so that merging extensions preserves each line's comment.
765      Type2::TextValue { value, .. } => {
766        let variant_name = to_pascal_case(value);
767        let original = value.to_string();
768        let serde_rename = if variant_name == original {
769          None
770        } else {
771          Some(original)
772        };
773        Ok(Some(RustTypeDef::Enum {
774          name,
775          variants: vec![RustEnumVariant {
776            name: variant_name,
777            inner_type: None,
778            serde_rename,
779            docs: docs.clone(),
780          }],
781          docs,
782        }))
783      }
784      Type2::IntValue { value, .. } => Ok(Some(RustTypeDef::Enum {
785        name,
786        variants: vec![RustEnumVariant {
787          name: if *value < 0 {
788            format!("Neg{}", value.unsigned_abs())
789          } else {
790            format!("N{}", value)
791          },
792          inner_type: None,
793          serde_rename: None,
794          docs: docs.clone(),
795        }],
796        docs,
797      })),
798      Type2::UintValue { value, .. } => Ok(Some(RustTypeDef::Enum {
799        name,
800        variants: vec![RustEnumVariant {
801          name: format!("N{}", value),
802          inner_type: None,
803          serde_rename: None,
804          docs: docs.clone(),
805        }],
806        docs,
807      })),
808      Type2::FloatValue { value, .. } => Ok(Some(RustTypeDef::Enum {
809        name,
810        variants: vec![RustEnumVariant {
811          name: format!("F{}", value.to_string().replace(['.', '-'], "_")),
812          inner_type: None,
813          serde_rename: None,
814          docs: docs.clone(),
815        }],
816        docs,
817      })),
818      Type2::ChoiceFromInlineGroup { group, .. } => {
819        let variants = group_to_enum_variants(group)?;
820        Ok(Some(RustTypeDef::Enum {
821          name,
822          variants,
823          docs,
824        }))
825      }
826      Type2::ChoiceFromGroup { ident, .. } => {
827        let target = to_pascal_case(ident.ident);
828        Ok(Some(RustTypeDef::TypeAlias { name, target, docs }))
829      }
830      Type2::TaggedData { t, .. } => {
831        let target = type_to_rust_string(t)?;
832        Ok(Some(RustTypeDef::TypeAlias { name, target, docs }))
833      }
834      _ => Ok(None),
835    }
836  } else {
837    Ok(None)
838  }
839}

844    Type2::UintValue { .. } => "u64".to_string(),
845    Type2::FloatValue { .. } => "f64".to_string(),
846    Type2::Typename { ident, .. } => cddl_ident_to_rust_type(ident.ident),
847    _ => "i64".to_string(),
848  }

868fn type_choice_to_variant(tc: &TypeChoice<'_>) -> Result<RustEnumVariant, CodegenError> {
869  let type1 = &tc.type1;
870  match &type1.type2 {
871    Type2::Typename { ident, .. } => {
872      let ident_str = ident.ident;
873      let variant_name = to_pascal_case(ident_str);
874      let inner = if is_prelude_type(ident_str) {
875        Some(cddl_ident_to_rust_type(ident_str))
876      } else {
877        Some(variant_name.clone())
878      };
879      Ok(RustEnumVariant {
880        name: variant_name,
881        inner_type: inner,
882        serde_rename: None,
883        docs: Vec::new(),
884      })
885    }
886    Type2::TextValue { value, .. } => {
887      let variant_name = to_pascal_case(value);
888      let original = value.to_string();
889      let serde_rename = if variant_name == original {
890        None
891      } else {
892        Some(original)
893      };
894      Ok(RustEnumVariant {
895        name: variant_name,
896        inner_type: None,
897        serde_rename,
898        docs: Vec::new(),
899      })
900    }
901    Type2::IntValue { value, .. } => {
902      let variant_name = if *value < 0 {
903        format!("Neg{}", value.unsigned_abs())
904      } else {
905        format!("N{}", value)
906      };
907      Ok(RustEnumVariant {
908        name: variant_name,
909        inner_type: None,
910        serde_rename: None,
911        docs: Vec::new(),
912      })
913    }
914    Type2::UintValue { value, .. } => Ok(RustEnumVariant {
915      name: format!("N{}", value),
916      inner_type: None,
917      serde_rename: None,
918      docs: Vec::new(),
919    }),
920    Type2::FloatValue { value, .. } => Ok(RustEnumVariant {
921      name: format!("F{}", value.to_string().replace(['.', '-'], "_")),
922      inner_type: None,
923      serde_rename: None,
924      docs: Vec::new(),
925    }),
926    Type2::Map { group, .. } => {
927      let fields = group_to_fields(group, &CommentIndex::empty())?;
928      let variant_name = if fields.is_empty() {
929        "Empty".to_string()
930      } else {
931        fields
932          .iter()
933          .map(|f| to_pascal_case(&f.original_name))
934          .collect::<Vec<_>>()
935          .join("")
936      };
937      Ok(RustEnumVariant {
938        name: variant_name,
939        inner_type: None,
940        serde_rename: None,
941        docs: Vec::new(),
942      })
943    }
944    Type2::Array { .. } => Ok(RustEnumVariant {
945      name: "Array".to_string(),
946      inner_type: Some("Vec<()>".to_string()),
947      serde_rename: None,
948      docs: Vec::new(),
949    }),
950    Type2::ParenthesizedType { pt, .. } => {
951      let rust_type = type_to_rust_string(pt)?;
952      let variant_name = to_pascal_case(&rust_type);
953      Ok(RustEnumVariant {
954        name: variant_name,
955        inner_type: Some(rust_type),
956        serde_rename: None,
957        docs: Vec::new(),
958      })
959    }
960    _ => Ok(RustEnumVariant {
961      name: "Unknown".to_string(),
962      inner_type: None,
963      serde_rename: None,
964      docs: Vec::new(),
965    }),
966  }
967}

994fn group_entry_to_fields(
995  entry: &GroupEntry<'_>,
996  comments: &CommentIndex<'_>,
997) -> Result<Option<Vec<RustField>>, CodegenError> {
998  match entry {
999    GroupEntry::ValueMemberKey { ge, span, .. } => {
1000      let docs = field_docs_for(comments, span);
1001      if let Some(field) = value_member_key_to_field(ge, docs)? {
1002        Ok(Some(vec![field]))
1003      } else {
1004        Ok(None)
1005      }
1006    }
1007    GroupEntry::TypeGroupname { ge, span, .. } => {
1008      let ident = ge.name.ident;
1009      let rust_type = cddl_ident_to_rust_type(ident);
1010      let field_name = to_snake_case(ident);
1011      let is_optional = ge
1012        .occur
1013        .as_ref()
1014        .map(|o| matches!(o.occur, Occur::Optional { .. }))
1015        .unwrap_or(false);
1016      let docs = field_docs_for(comments, span);
1017      Ok(Some(vec![RustField {
1018        name: field_name,
1019        original_name: ident.to_string(),
1020        rust_type,
1021        is_optional,
1022        docs,
1023      }]))
1024    }
1025    GroupEntry::InlineGroup { group, occur, .. } => {
1026      let is_optional = occur
1027        .as_ref()
1028        .map(|o| matches!(o.occur, Occur::Optional { .. }))
1029        .unwrap_or(false);
1030      let mut fields = group_to_fields(group, comments)?;
1031      if is_optional {
1032        for f in &mut fields {
1033          f.is_optional = true;
1034        }
1035      }
1036      Ok(Some(fields))
1037    }
1038  }
1039}

1041fn field_docs_for(comments: &CommentIndex<'_>, span: &cddl::ast::Span) -> Vec<String> {
1042  let line = span_line(span);
1043  let mut docs: Vec<String> = comments
1044    .preceding(line)
1045    .into_iter()
1046    .map(String::from)
1047    .collect();
1048  if let Some(trailing) = comments.trailing(line) {
1049    docs.push(trailing.to_string());
1050  }
1051  docs
1052}

1054fn value_member_key_to_field(
1055  vmke: &ValueMemberKeyEntry<'_>,
1056  docs: Vec<String>,
1057) -> Result<Option<RustField>, CodegenError> {
1058  let (field_name, original_name) = match &vmke.member_key {
1059    Some(MemberKey::Bareword { ident, .. }) => {
1060      (to_snake_case(ident.ident), ident.ident.to_string())
1061    }
1062    Some(MemberKey::Value { value, .. }) => {
1063      let s = value.to_string();
1064      let s = s.trim_matches('"');
1065      (to_snake_case(s), s.to_string())
1066    }
1067    Some(MemberKey::Type1 { t1, .. }) => {
1068      let key_type = type1_to_rust_string(t1)?;
1069      let value_type = type_to_rust_string(&vmke.entry_type)?;
1070      let rust_type = format!("std::collections::HashMap<{}, {}>", key_type, value_type);
1071      return Ok(Some(RustField {
1072        name: "entries".to_string(),
1073        original_name: "entries".to_string(),
1074        rust_type,
1075        is_optional: false,
1076        docs,
1077      }));
1078    }
1079    None => {
1080      let rust_type = type_to_rust_string(&vmke.entry_type)?;
1081      return Ok(Some(RustField {
1082        name: "value".to_string(),
1083        original_name: "value".to_string(),
1084        rust_type,
1085        is_optional: false,
1086        docs,
1087      }));
1088    }
1089    _ => return Ok(None),
1090  };
1091
1092  let is_optional = vmke
1093    .occur
1094    .as_ref()
1095    .map(|o| matches!(o.occur, Occur::Optional { .. }))
1096    .unwrap_or(false);
1097
1098  let is_vec = vmke
1099    .occur
1100    .as_ref()
1101    .map(|o| is_vec_occurrence(&o.occur))
1102    .unwrap_or(false);
1103
1104  let rust_type = type_to_rust_string(&vmke.entry_type)?;
1105
1106  let final_type = if is_vec {
1107    format!("Vec<{}>", rust_type)
1108  } else {
1109    rust_type
1110  };
1111
1112  Ok(Some(RustField {
1113    name: field_name,
1114    original_name,
1115    rust_type: final_type,
1116    is_optional,
1117    docs,
1118  }))
1119}

1122  match occur {
1123    Occur::ZeroOrMore { .. } | Occur::OneOrMore { .. } => true,
1124    Occur::Exact { upper: Some(u), .. } => *u > 1,
1125    _ => false,
1126  }

1129fn array_group_to_type(group: &Group<'_>) -> Result<String, CodegenError> {
1130  if group.group_choices.len() == 1 {
1131    let gc = &group.group_choices[0];
1132    if gc.group_entries.len() == 1 {
1133      let (entry, _) = &gc.group_entries[0];
1134      match entry {
1135        GroupEntry::ValueMemberKey { ge, .. } => {
1136          let element_type = type_to_rust_string(&ge.entry_type)?;
1137          return Ok(format!("Vec<{}>", element_type));
1138        }
1139        GroupEntry::TypeGroupname { ge, .. } => {
1140          let element_type = cddl_ident_to_rust_type(ge.name.ident);
1141          return Ok(format!("Vec<{}>", element_type));
1142        }
1143        _ => {}
1144      }
1145    }
1146    if !gc.group_entries.is_empty() {
1147      let mut types = Vec::new();
1148      for (entry, _) in &gc.group_entries {
1149        match entry {
1150          GroupEntry::ValueMemberKey { ge, .. } => {
1151            types.push(type_to_rust_string(&ge.entry_type)?);
1152          }
1153          GroupEntry::TypeGroupname { ge, .. } => {
1154            types.push(cddl_ident_to_rust_type(ge.name.ident));
1155          }
1156          _ => types.push("()".to_string()),
1157        }
1158      }
1159      if types.len() == 1 {
1160        return Ok(format!("Vec<{}>", types[0]));
1161      }
1162      return Ok(format!("({})", types.join(", ")));
1163    }
1164  }
1165  Ok("Vec<()>".to_string())
1166}

1168fn type_to_rust_string(ty: &Type<'_>) -> Result<String, CodegenError> {
1169  if ty.type_choices.len() == 1 {
1170    return type1_to_rust_string(&ty.type_choices[0].type1);
1171  }
1172  if ty.type_choices.len() == 2 {
1173    let (a, b) = (&ty.type_choices[0].type1, &ty.type_choices[1].type1);
1174    if is_null_type(&b.type2) {
1175      let inner = type1_to_rust_string(a)?;
1176      return Ok(format!("Option<{}>", inner));
1177    }
1178    if is_null_type(&a.type2) {
1179      let inner = type1_to_rust_string(b)?;
1180      return Ok(format!("Option<{}>", inner));
1181    }
1182  }
1183  Ok("ciborium::Value".to_string())
1184}

1190fn type2_to_rust_string(type2: &Type2<'_>) -> Result<String, CodegenError> {
1191  match type2 {
1192    Type2::Typename { ident, .. } => Ok(cddl_ident_to_rust_type(ident.ident)),
1193    Type2::Map { group, .. } => {
1194      if let Some(field) = detect_table_type(group)? {
1195        Ok(field)
1196      } else {
1197        Ok("ciborium::Value".to_string())
1198      }
1199    }
1200    Type2::Array { group, .. } => array_group_to_type(group),
1201    Type2::TextValue { .. } => Ok("String".to_string()),
1202    Type2::IntValue { .. } => Ok("i64".to_string()),
1203    Type2::UintValue { .. } => Ok("u64".to_string()),
1204    Type2::FloatValue { .. } => Ok("f64".to_string()),
1205    Type2::UTF8ByteString { .. } | Type2::B16ByteString { .. } | Type2::B64ByteString { .. } => {
1206      // CDDL byte strings -> CBOR major type 2. Emitted as `Vec<u8>`;
1207      // the struct renderer pairs each such field with a
1208      // `#[serde_as(as = "serde_with::Bytes")]` attribute so serde
1209      // hits the byte-string branch instead of the default
1210      // array-of-u8 encoding.
1211      Ok("Vec<u8>".to_string())
1212    }
1213    Type2::ParenthesizedType { pt, .. } => type_to_rust_string(pt),
1214    Type2::Unwrap { ident, .. } => Ok(to_pascal_case(ident.ident)),
1215    Type2::TaggedData { t, .. } => type_to_rust_string(t),
1216    Type2::Any { .. } => Ok("ciborium::Value".to_string()),
1217    Type2::ChoiceFromInlineGroup { .. } | Type2::ChoiceFromGroup { .. } => {
1218      Ok("ciborium::Value".to_string())
1219    }
1220    Type2::DataMajorType { mt, .. } => Ok(major_type_to_rust(*mt)),
1221  }
1222}

1224fn detect_table_type(group: &Group<'_>) -> Result<Option<String>, CodegenError> {
1225  if group.group_choices.len() != 1 {
1226    return Ok(None);
1227  }
1228  let gc = &group.group_choices[0];
1229  if gc.group_entries.len() != 1 {
1230    return Ok(None);
1231  }
1232  let (entry, _) = &gc.group_entries[0];
1233  if let GroupEntry::ValueMemberKey { ge, .. } = entry {
1234    if let Some(MemberKey::Type1 { t1, .. }) = &ge.member_key {
1235      let key_type = type1_to_rust_string(t1)?;
1236      let value_type = type_to_rust_string(&ge.entry_type)?;
1237      return Ok(Some(format!(
1238        "std::collections::HashMap<{}, {}>",
1239        key_type, value_type
1240      )));
1241    }
1242  }
1243  Ok(None)
1244}

1305/// macro's call site. Rules in this table are not emitted as generated
1306/// type aliases or structs (the consumer provides the real definition).
1307const RULE_SUBSTITUTIONS: &[(&str, &str)] = &[
1308  ("jumbf-uri-type", "crate::jumbf_uri::JumbfUri"),
1309];

1311pub(crate) fn rule_substitution(cddl_ident: &str) -> Option<&'static str> {
1312  RULE_SUBSTITUTIONS
1313    .iter()
1314    .find(|(k, _)| *k == cddl_ident)
1315    .map(|(_, v)| *v)
1316}

1395fn render_type_defs(defs: &[RustTypeDef]) -> Result<String, CodegenError> {
1396  let mut output = String::new();
1397
1398  for (idx, def) in defs.iter().enumerate() {
1399    if idx > 0 {
1400      output.push('\n');
1401    }
1402    match def {
1403      RustTypeDef::Struct { name, fields, docs } => {
1404        render_struct(&mut output, name, fields, docs)?;
1405      }
1406      RustTypeDef::TypeAlias { name, target, docs } => {
1407        render_type_alias(&mut output, name, target, docs)?;
1408      }
1409      RustTypeDef::Enum {
1410        name,
1411        variants,
1412        docs,
1413      } => {
1414        render_enum(&mut output, name, variants, docs)?;
1415      }
1416    }
1417  }
1418
1419  Ok(output)
1420}

1422fn render_docs(output: &mut String, indent: &str, docs: &[String]) -> Result<(), CodegenError> {
1423  for line in docs {
1424    for sub in line.split('\n') {
1425      // Rust's doc-comment parser is lenient, but writing the slash-star
1426      // sequence inside a `///` line is still fine. We trim trailing
1427      // whitespace to keep the generated output tidy.
1428      let trimmed = sub.trim_end();
1429      if trimmed.is_empty() {
1430        writeln!(output, "{}///", indent)?;
1431      } else {
1432        writeln!(output, "{}/// {}", indent, trimmed)?;
1433      }
1434    }
1435  }
1436  Ok(())
1437}

1439fn render_struct(
1440  output: &mut String,
1441  name: &str,
1442  fields: &[RustField],
1443  docs: &[String],
1444) -> Result<(), CodegenError> {
1445  // Precompute each field's full rendered type (including Option wrap)
1446  // and its serde_as pattern, so we know whether to emit the
1447  // struct-level `#[serde_with::serde_as]` attribute.
1448  let field_types: Vec<String> = fields
1449    .iter()
1450    .map(|f| {
1451      if f.is_optional {
1452        format!("Option<{}>", f.rust_type)
1453      } else {
1454        f.rust_type.clone()
1455      }
1456    })
1457    .collect();
1458  let field_patterns: Vec<Option<String>> = field_types
1459    .iter()
1460    .map(|ty| bytes_serde_as_pattern(ty))
1461    .collect();
1462  let needs_serde_as = field_patterns.iter().any(Option::is_some);
1463
1464  render_docs(output, "", docs)?;
1465  if needs_serde_as {
1466    writeln!(output, "#[serde_with::serde_as]")?;
1467  }
1468  writeln!(
1469    output,
1470    "#[derive(Clone, Debug, serde::Deserialize, serde::Serialize)]"
1471  )?;
1472  writeln!(output, "pub struct {} {{", name)?;
1473  for (idx, field) in fields.iter().enumerate() {
1474    render_docs(output, "    ", &field.docs)?;
1475    if let Some(pattern) = &field_patterns[idx] {
1476      writeln!(output, "    #[serde_as(as = \"{}\")]", pattern)?;
1477    }
1478    if field.name != field.original_name {
1479      writeln!(output, "    #[serde(rename = \"{}\")]", field.original_name)?;
1480    }
1481    if field.is_optional {
1482      writeln!(
1483        output,
1484        "    #[serde(skip_serializing_if = \"Option::is_none\")]"
1485      )?;
1486    }
1487    writeln!(
1488      output,
1489      "    pub {}: {},",
1490      field.name, field_types[idx]
1491    )?;
1492  }
1493  writeln!(output, "}}")?;
1494  Ok(())
1495}

1497fn render_type_alias(
1498  output: &mut String,
1499  name: &str,
1500  target: &str,
1501  docs: &[String],
1502) -> Result<(), CodegenError> {
1503  render_docs(output, "", docs)?;
1504  // A plain `pub type X = Vec<u8>` would inherit `Vec<u8>`'s default
1505  // serde behaviour (array of u8), which is the wrong CBOR major
1506  // type. Emit a newtype so the bytes attribute can attach to the
1507  // inner field.
1508  if let Some(pattern) = bytes_serde_as_pattern(target) {
1509    writeln!(output, "#[serde_with::serde_as]")?;
1510    writeln!(
1511      output,
1512      "#[derive(Clone, Debug, serde::Deserialize, serde::Serialize)]"
1513    )?;
1514    writeln!(output, "#[serde(transparent)]")?;
1515    writeln!(
1516      output,
1517      "pub struct {}(#[serde_as(as = \"{}\")] pub {});",
1518      name, pattern, target
1519    )?;
1520  } else {
1521    writeln!(output, "pub type {} = {};", name, target)?;
1522  }
1523  Ok(())
1524}

1526fn render_enum(
1527  output: &mut String,
1528  name: &str,
1529  variants: &[RustEnumVariant],
1530  docs: &[String],
1531) -> Result<(), CodegenError> {
1532  let variant_patterns: Vec<Option<String>> = variants
1533    .iter()
1534    .map(|v| v.inner_type.as_deref().and_then(bytes_serde_as_pattern))
1535    .collect();
1536  let needs_serde_as = variant_patterns.iter().any(Option::is_some);
1537
1538  render_docs(output, "", docs)?;
1539  if needs_serde_as {
1540    writeln!(output, "#[serde_with::serde_as]")?;
1541  }
1542  writeln!(
1543    output,
1544    "#[derive(Clone, Debug, serde::Deserialize, serde::Serialize)]"
1545  )?;
1546  writeln!(output, "#[serde(untagged)]")?;
1547  writeln!(output, "pub enum {} {{", name)?;
1548  for (idx, variant) in variants.iter().enumerate() {
1549    render_docs(output, "    ", &variant.docs)?;
1550    if let Some(rename) = &variant.serde_rename {
1551      writeln!(
1552        output,
1553        "    #[serde(rename = \"{}\")]",
1554        rename.replace('\\', "\\\\").replace('"', "\\\"")
1555      )?;
1556    }
1557    if let Some(inner) = &variant.inner_type {
1558      if let Some(pattern) = &variant_patterns[idx] {
1559        writeln!(
1560          output,
1561          "    {}(#[serde_as(as = \"{}\")] {}),",
1562          variant.name, pattern, inner
1563        )?;
1564      } else {
1565        writeln!(output, "    {}({}),", variant.name, inner)?;
1566      }
1567    } else {
1568      writeln!(output, "    {},", variant.name)?;
1569    }
1570  }
1571  writeln!(output, "}}")?;
1572  Ok(())
1573}

1575/// If `ty` references `Vec<u8>` anywhere (directly or nested inside
1576/// `Vec`, `Option`, `HashMap`, `BTreeMap`), return a matching
1577/// `serde_with` `as = "..."` pattern. Returns `None` when no bytes
1578/// substitution is needed.
1579///
1580/// - `Vec<u8>` → `serde_with::Bytes`
1581/// - `Vec<Vec<u8>>` → `Vec<serde_with::Bytes>`
1582/// - `Option<Vec<u8>>` → `Option<serde_with::Bytes>`
1583/// - `std::collections::HashMap<K, Vec<u8>>` → `std::collections::HashMap<_, serde_with::Bytes>`
1584fn bytes_serde_as_pattern(ty: &str) -> Option<String> {
1585  let ty = ty.trim();
1586  if ty == "Vec<u8>" {
1587    return Some("serde_with::Bytes".to_string());
1588  }
1589  for wrapper in &["Vec", "Option"] {
1590    if let Some(inner) = strip_generic(wrapper, ty) {
1591      if let Some(inner_pattern) = bytes_serde_as_pattern(inner) {
1592        return Some(format!("{}<{}>", wrapper, inner_pattern));
1593      }
1594    }
1595  }
1596  for map in &[
1597    "std::collections::HashMap",
1598    "std::collections::BTreeMap",
1599    "HashMap",
1600    "BTreeMap",
1601  ] {
1602    if let Some(inner) = strip_generic(map, ty) {
1603      if let Some((_key, value)) = split_top_level_comma(inner) {
1604        if let Some(value_pattern) = bytes_serde_as_pattern(value.trim()) {
1605          return Some(format!("{}<_, {}>", map, value_pattern));
1606        }
1607      }
1608    }
1609  }
1610  None
1611}

1613/// Strip `Name<...>` wrapping and return the inner substring, or `None`
1614/// if `ty` isn't of that form.
1615fn strip_generic<'a>(name: &str, ty: &'a str) -> Option<&'a str> {
1616  let rest = ty.strip_prefix(name)?;
1617  let rest = rest.strip_prefix('<')?;
1618  let rest = rest.strip_suffix('>')?;
1619  Some(rest)
1620}
1621
1622/// Split `s` at the first top-level (bracket-depth-zero) comma.
1623fn split_top_level_comma(s: &str) -> Option<(&str, &str)> {
1624  let mut depth = 0i32;
1625  for (i, c) in s.char_indices() {
1626    match c {
1627      '<' | '(' | '[' => depth += 1,
1628      '>' | ')' | ']' => depth -= 1,
1629      ',' if depth == 0 => return Some((&s[..i], &s[i + 1..])),
1630      _ => {}
1631    }
1632  }
1633  None
1634}

1636pub(crate) fn to_pascal_case(s: &str) -> String {
1637  // Split into chunks on explicit separators and on lowercase->uppercase
1638  // transitions, then title-case each chunk. This lets ALL-CAPS CDDL
1639  // rule names like `EXCLUSION_RANGE-map` collapse to `ExclusionRangeMap`
1640  // while camelCase inputs like `validationResults` still break at the
1641  // `R` and yield `ValidationResults`.
1642  let mut chunks: Vec<String> = Vec::new();
1643  let mut current = String::new();
1644  let mut prev_was_lower = false;
1645  for c in s.chars() {
1646    if matches!(c, '-' | '_' | '.' | ' ' | ':' | '/' | '@') {
1647      if !current.is_empty() {
1648        chunks.push(std::mem::take(&mut current));
1649      }
1650      prev_was_lower = false;
1651    } else if c.is_ascii_alphanumeric() {
1652      if c.is_ascii_uppercase() && prev_was_lower && !current.is_empty() {
1653        chunks.push(std::mem::take(&mut current));
1654      }
1655      current.push(c);
1656      prev_was_lower = c.is_ascii_lowercase();
1657    } else {
1658      // Any other punctuation acts as a separator.
1659      if !current.is_empty() {
1660        chunks.push(std::mem::take(&mut current));
1661      }
1662      prev_was_lower = false;
1663    }
1664  }
1665  if !current.is_empty() {
1666    chunks.push(current);
1667  }
1668
1669  let mut result = String::new();
1670  for chunk in chunks {
1671    let mut chars = chunk.chars();
1672    if let Some(first) = chars.next() {
1673      result.extend(first.to_uppercase());
1674      for c in chars {
1675        result.extend(c.to_lowercase());
1676      }
1677    }
1678  }
1679
1680  if let Some(first) = result.chars().next() {
1681    if first.is_ascii_digit() {
1682      result.insert(0, '_');
1683    }
1684  }
1685  if result.is_empty() {
1686    return "Unknown".to_string();
1687  }
1688  result
1689}

1691fn to_snake_case(s: &str) -> String {
1692  let mut result = String::with_capacity(s.len() + 4);
1693  let mut prev_was_upper = false;
1694  let mut prev_was_separator = false;
1695  for (i, c) in s.chars().enumerate() {
1696    if c == '-' || c == '.' || c == ' ' || c == ':' || c == '/' || c == '@' {
1697      if !result.is_empty() && !prev_was_separator {
1698        result.push('_');
1699      }
1700      prev_was_separator = true;
1701      prev_was_upper = false;
1702    } else if c.is_uppercase() {
1703      if i > 0 && !prev_was_upper && !prev_was_separator {
1704        result.push('_');
1705      }
1706      result.push(c.to_lowercase().next().unwrap());
1707      prev_was_upper = true;
1708      prev_was_separator = false;
1709    } else if c.is_ascii_alphanumeric() || c == '_' {
1710      result.push(c);
1711      prev_was_upper = false;
1712      // Treat a literal underscore as a separator so a following upper
1713      // case letter does not trigger a second insert (e.g. the CDDL
1714      // key `informational_URI` should snake-case to `informational_uri`,
1715      // not `informational__uri`).
1716      prev_was_separator = c == '_';
1717    } else {
1718      // Drop any other punctuation so we end up with a valid Rust ident.
1719      if !result.is_empty() && !prev_was_separator {
1720        result.push('_');
1721      }
1722      prev_was_separator = true;
1723      prev_was_upper = false;
1724    }
1725  }
1726  // Trim trailing underscores that resulted from the separator handling.
1727  while result.ends_with('_') && result.len() > 1 {
1728    result.pop();
1729  }
1730  // Ensure the identifier starts with a letter or underscore (Rust rule).
1731  if let Some(first) = result.chars().next() {
1732    if first.is_ascii_digit() {
1733      result.insert(0, '_');
1734    }
1735  }
1736  if is_rust_keyword(&result) {
1737    result.push('_');
1738  }
1739  if result.is_empty() {
1740    return "value".to_string();
1741  }
1742  result
1743}
