1//! `refresh-spec`: regenerates the committed Rust sources in `c2pa-spec`
2//! from the CDDL / YAML / ABNF files that accompany the C2PA technical
3//! specification.
4//!
5//! Typical workflow when bumping the spec:
6//!
7//! ```text
8//! cargo run -p c2pa-spec-codegen -- --download
9//! cargo build -p c2pa-spec
10//! git diff c2pa-spec/
11//! ```

22/// Specifications index on `spec.c2pa.org`. The page is an HTML
23/// redirect that names the latest version in its canonical link and
24/// meta-refresh target (e.g.
25/// `.../specifications/specifications/2.4/index.html`); we scrape the
26/// version out of there to build the schemas-bundle URL for the current
27/// spec.
28const SPEC_INDEX_URL: &str = "https://spec.c2pa.org/specifications/";
29
30/// CDDL files that fail to parse with `cddl` 0.10.5 and are skipped
31/// when building the master schema. Drop a corrected copy into
32/// `c2pa-spec/schema-patches/` to un-exclude a file. See
33/// `c2pa-spec/CDDL_ISSUES.md`.
34const EXCLUDED_CDDLS: &[&str] = &[
35    "jsonld.cddl",
36    // Self-reference: this is the master file the codegen writes.
37    "c2pa.cddl",
38];
39
40/// Regenerate c2pa-spec source files from the CDDL, YAML, and ABNF
41/// schemas that accompany the C2PA technical specification.
42#[derive(Parser, Debug)]
43#[command(version, about, long_about = None)]
44struct Cli {
45    /// Download the schemas bundle first, overwriting the vendored
46    /// CDDL / YAML / ABNF files in `--spec-dir`.
47    #[arg(short, long)]
48    download: bool,
49
50    /// Schemas bundle URL. Defaults to whichever version
51    /// `https://spec.c2pa.org` currently redirects to.
52    #[arg(long)]
53    url: Option<String>,
54
55    /// Path to the `c2pa-spec` crate root. Defaults to the sibling of
56    /// this crate inside the workspace.
57    #[arg(long, default_value_os_t = default_spec_dir())]
58    spec_dir: PathBuf,
59
60    /// Directory of manually patched CDDL files that override the
61    /// downloaded / vendored versions. Any `<patches-dir>/foo.cddl`
62    /// replaces `<spec-dir>/schemas/cddl/foo.cddl` when building the
63    /// master schema, and automatically un-excludes its file from
64    /// `EXCLUDED_CDDLS`. Defaults to `<spec-dir>/schema-patches/`.
65    #[arg(long)]
66    patches_dir: Option<PathBuf>,
67}
68
69fn default_spec_dir() -> PathBuf {
70    PathBuf::from(env!("CARGO_MANIFEST_DIR"))
71        .parent()
72        .expect("codegen manifest should have a parent workspace")
73        .join("c2pa-spec")
74}

76/// Discover the schemas bundle URL by fetching the `spec.c2pa.org`
77/// specifications index (which is a static HTML page with a
78/// meta-refresh / canonical link pointing at the current versioned
79/// index), extracting the version segment from the body, and pasting it
80/// into the known bundle path template.
81fn resolve_latest_spec_url() -> Result<String, String> {
82    let body = ureq::get(SPEC_INDEX_URL)
83        .call()
84        .map_err(|e| format!("GET {SPEC_INDEX_URL}: {e}"))?
85        .into_body()
86        .read_to_string()
87        .map_err(|e| format!("read body of {SPEC_INDEX_URL}: {e}"))?;
88
89    let version = extract_spec_version(&body).ok_or_else(|| {
90        format!("no `/specifications/specifications/<version>/` link found at {SPEC_INDEX_URL}")
91    })?;
92
93    Ok(format!(
94        "https://spec.c2pa.org/specifications/specifications/{version}/specs/_attachments/C2PA_Schemas.zip"
95    ))
96}
97
98/// Pull the first `specifications/specifications/<VERSION>/` version
99/// segment out of the given HTML body. Accepts `<VERSION>` as a sequence
100/// of digits and dots (e.g. `2`, `2.4`, `2.4.1`).
101fn extract_spec_version(body: &str) -> Option<String> {
102    let marker = "/specifications/specifications/";
103    let mut cursor = 0;
104    while let Some(rel) = body[cursor..].find(marker) {
105        let start = cursor + rel + marker.len();
106        let tail = &body[start..];
107        let end = tail.find('/')?;
108        let candidate = &tail[..end];
109        if !candidate.is_empty() && candidate.chars().all(|c| c.is_ascii_digit() || c == '.') {
110            return Some(candidate.to_string());
111        }
112        cursor = start + end;
113    }
114    None
115}

151fn main() {
152    let cli = Cli::parse();
153    if !cli.spec_dir.is_dir() {
154        fail(&format!(
155            "spec dir {} does not exist",
156            cli.spec_dir.display()
157        ));
158    }
159
160    let schemas_dir = cli.spec_dir.join("schemas");
161    let cddl_dir = schemas_dir.join("cddl");
162    let patches_dir = cli
163        .patches_dir
164        .clone()
165        .unwrap_or_else(|| cli.spec_dir.join("schema-patches"));
166    let metadata_yaml = schemas_dir.join("valid_metadata_fields.yml");
167    let urn_abnf = schemas_dir.join("c2pa_urn.abnf");
168    let generated_rs = cli.spec_dir.join("src/generated.rs");
169    let fields_rs = cli.spec_dir.join("src/valid_metadata_fields.rs");
170    let cargo_toml = cli.spec_dir.join("Cargo.toml");
171
172    // Record the spec version we start from so the workflow can tell the
173    // user about it. The version lives in SemVer build metadata on
174    // `c2pa-spec`'s crate version (`0.1.0+spec-2.4`).
175    let old_spec = read_spec_version(&cargo_toml);
176
177    if cli.download {
178        let url = match cli.url {
179            Some(u) => u,
180            None => {
181                let resolved = resolve_latest_spec_url()
182                    .unwrap_or_else(|e| fail(&format!("could not resolve latest spec URL: {e}")));
183                eprintln!("refresh-spec: resolved latest spec URL to {resolved}");
184                resolved
185            }
186        };
187        let new_spec = extract_version_from_url(&url);
188        refresh_from_spec(&url, &cddl_dir, &metadata_yaml, &urn_abnf);
189
190        if let Some(ver) = &new_spec {
191            if let Err(e) = write_spec_version(&cargo_toml, ver) {
192                fail(&format!("update Cargo.toml spec version: {e}"));
193            }
194        }
195
196        // Emit structured lines the refresh workflow parses to build
197        // the PR title.
198        eprintln!(
199            "refresh-spec: old_spec_version={}",
200            old_spec.as_deref().unwrap_or("unknown")
201        );
202        eprintln!(
203            "refresh-spec: new_spec_version={}",
204            new_spec.as_deref().unwrap_or("unknown")
205        );
206    }
207
208    let master = build_master_cddl(&cddl_dir, &patches_dir);
209    let master_path = cddl_dir.join("c2pa.cddl");
210    fs::write(&master_path, &master)
211        .unwrap_or_else(|e| fail(&format!("write {}: {e}", master_path.display())));
212    eprintln!(
213        "refresh-spec: wrote {}",
214        relative(&master_path, &cli.spec_dir)
215    );
216
217    emit_generated_rs(&master, &generated_rs);
218    eprintln!(
219        "refresh-spec: wrote {}",
220        relative(&generated_rs, &cli.spec_dir)
221    );
222
223    emit_valid_metadata_fields(&metadata_yaml, &fields_rs);
224    eprintln!(
225        "refresh-spec: wrote {}",
226        relative(&fields_rs, &cli.spec_dir)
227    );
228}

230fn relative(path: &Path, base: &Path) -> String {
231    path.strip_prefix(base)
232        .map(|p| p.display().to_string())
233        .unwrap_or_else(|_| path.display().to_string())
234}

236/// Concatenate every `.cddl` file in `cddl_dir` (except the excluded
237/// ones and the generated master) into a single string. Drops duplicate
238/// `name = ...` rule definitions so `font-weight-range` doesn't appear
239/// twice. If `patches_dir` exists, any `<patches_dir>/foo.cddl` replaces
240/// `<cddl_dir>/foo.cddl` in the output and un-excludes that filename
241/// from `EXCLUDED_CDDLS`.
242fn build_master_cddl(cddl_dir: &Path, patches_dir: &Path) -> String {
243    let patched = collect_patch_names(patches_dir);
244    for name in &patched {
245        eprintln!("refresh-spec: applying patch {name}");
246    }
247
248    let mut entries: Vec<PathBuf> = fs::read_dir(cddl_dir)
249        .unwrap_or_else(|e| panic!("read {}: {e}", cddl_dir.display()))
250        .filter_map(Result::ok)
251        .map(|e| e.path())
252        .filter(|p| p.extension().is_some_and(|e| e == "cddl"))
253        .filter(|p| {
254            let name = p.file_name().unwrap().to_str().unwrap();
255            patched.contains(name) || !EXCLUDED_CDDLS.contains(&name)
256        })
257        .collect();
258    entries.sort();
259
260    // Warn about patches that don't correspond to any file in cddl_dir
261    // so typos / drift from upstream surface instead of silently adding
262    // a new file to the master schema.
263    let cddl_names: HashSet<String> = entries
264        .iter()
265        .map(|p| p.file_name().unwrap().to_string_lossy().into_owned())
266        .collect();
267    for name in &patched {
268        if !cddl_names.contains(name) {
269            eprintln!(
270                "refresh-spec: warning: patch {name} has no matching file in {}",
271                cddl_dir.display()
272            );
273        }
274    }
275
276    let excluded_report: Vec<&str> = EXCLUDED_CDDLS[..EXCLUDED_CDDLS.len() - 1]
277        .iter()
278        .filter(|name| !patched.contains(**name))
279        .copied()
280        .collect();
281
282    let mut out = String::new();
283    out.push_str("; C2PA master CDDL schema\n");
284    out.push_str("; Auto-generated by `cargo run -p c2pa-spec-codegen -- --help`.\n");
285    out.push_str("; Do not edit by hand.\n");
286    if !excluded_report.is_empty() {
287        out.push_str("; Excluded files (see CDDL_ISSUES.md): ");
288        out.push_str(&excluded_report.join(", "));
289        out.push('\n');
290    }
291    if !patched.is_empty() {
292        let mut names: Vec<&str> = patched.iter().map(String::as_str).collect();
293        names.sort_unstable();
294        out.push_str("; Patched files (from schema-patches/): ");
295        out.push_str(&names.join(", "));
296        out.push('\n');
297    }
298    out.push('\n');
299
300    let mut seen_eq: HashSet<String> = HashSet::new();
301
302    for path in &entries {
303        let name = path.file_name().unwrap().to_string_lossy();
304        let source_path = if patched.contains(name.as_ref()) {
305            patches_dir.join(name.as_ref())
306        } else {
307            path.clone()
308        };
309        let contents = fs::read_to_string(&source_path)
310            .unwrap_or_else(|e| panic!("read {}: {e}", source_path.display()));
311        out.push_str("\n; ================================================================\n");
312        if patched.contains(name.as_ref()) {
313            out.push_str(&format!("; From {name} (patched)\n"));
314        } else {
315            out.push_str(&format!("; From {name}\n"));
316        }
317        out.push_str("; ================================================================\n");
318        // Blank line so the separator comments above do not attach as
319        // doc-comments to the first rule in the following file.
320        out.push('\n');
321        for line in contents.lines() {
322            if let Some(rule_name) = parse_eq_rule_name(line) {
323                if !seen_eq.insert(rule_name.to_string()) {
324                    out.push_str(&format!(
325                        "; [refresh-spec dedup] skipped duplicate `=` definition of {rule_name}\n"
326                    ));
327                    continue;
328                }
329            }
330            out.push_str(line);
331            out.push('\n');
332        }
333    }
334
335    out
336}

338/// Return the set of `*.cddl` file names present in `patches_dir`.
339/// A missing directory yields an empty set (patches are optional).
340fn collect_patch_names(patches_dir: &Path) -> HashSet<String> {
341    let Ok(read) = fs::read_dir(patches_dir) else {
342        return HashSet::new();
343    };
344    read.filter_map(Result::ok)
345        .map(|e| e.path())
346        .filter(|p| p.extension().is_some_and(|e| e == "cddl"))
347        .filter_map(|p| p.file_name().and_then(|s| s.to_str()).map(String::from))
348        .collect()
349}

351fn parse_eq_rule_name(line: &str) -> Option<&str> {
352    if line.is_empty() || line.starts_with(char::is_whitespace) {
353        return None;
354    }
355    let first = line.chars().next()?;
356    if !(first.is_ascii_alphabetic() || first == '$') {
357        return None;
358    }
359    let eq_pos = line.find('=')?;
360    if line[..eq_pos].ends_with('/') {
361        return None;
362    }
363    let before = line[..eq_pos].trim_end();
364    if before.contains(char::is_whitespace) {
365        return None;
366    }
367    Some(before)
368}

370/// Parse the master CDDL, run codegen, and write the output to
371/// `generated.rs` with the autogen header.
372fn emit_generated_rs(master_cddl: &str, dest: &Path) {
373    let cddl_ast = cddl::parser::cddl_from_str(master_cddl, true)
374        .unwrap_or_else(|e| panic!("parse master CDDL: {e}"));
375    let body = codegen::generate_all_types(&cddl_ast, master_cddl)
376        .unwrap_or_else(|e| panic!("codegen: {e}"));
377
378    let mut out = String::new();
379    out.push_str("// Auto-generated by `cargo run -p c2pa-spec-codegen`.\n");
380    out.push_str("// Do not edit by hand.\n\n");
381    out.push_str(&body);
382
383    fs::write(dest, out).unwrap_or_else(|e| panic!("write {}: {e}", dest.display()));
384}

386/// Parse `valid_metadata_fields.yml` and write the
387/// `pub mod valid_metadata_fields { ... }` contents to `dest`.
388fn emit_valid_metadata_fields(yaml_path: &Path, dest: &Path) {
389    let contents = fs::read_to_string(yaml_path)
390        .unwrap_or_else(|e| panic!("read {}: {e}", yaml_path.display()));
391    let groups = parse_metadata_yaml(&contents);
392
393    let mut out = String::new();
394    out.push_str(
395        "// Auto-generated by `cargo run -p c2pa-spec-codegen` from valid_metadata_fields.yml.\n",
396    );
397    out.push_str("// Do not edit by hand.\n\n");
398
399    for (name, fields) in &groups {
400        let const_name = to_upper_snake(name);
401        out.push_str(&format!(
402            "/// Valid metadata field names in the `{name}` group.\n"
403        ));
404        out.push_str(&format!("pub const {const_name}: &[&str] = &[\n"));
405        for field in fields {
406            out.push_str(&format!("    {},\n", rust_string_literal(field)));
407        }
408        out.push_str("];\n\n");
409    }
410
411    out.push_str("/// Every group as a `(group_name, fields)` pair, in source order.\n");
412    out.push_str("pub const ALL: &[(&str, &[&str])] = &[\n");
413    for (name, _) in &groups {
414        let const_name = to_upper_snake(name);
415        out.push_str(&format!(
416            "    ({}, {const_name}),\n",
417            rust_string_literal(name)
418        ));
419    }
420    out.push_str("];\n");
421
422    fs::write(dest, out).unwrap_or_else(|e| panic!("write {}: {e}", dest.display()));
423}

425/// Line-based parser for the subset of YAML used in
426/// `valid_metadata_fields.yml` (top-level `Group:` headers with
427/// `- field` list items). Returns groups in source order.
428fn parse_metadata_yaml(contents: &str) -> Vec<(String, Vec<String>)> {
429    let mut groups: Vec<(String, Vec<String>)> = Vec::new();
430    let mut current: Option<(String, Vec<String>)> = None;
431    for line in contents.lines() {
432        let trimmed = line.trim_start();
433        if trimmed.is_empty() || trimmed.starts_with('#') {
434            continue;
435        }
436        if let Some(rest) = trimmed.strip_prefix("- ") {
437            let value = rest.trim().to_string();
438            match current.as_mut() {
439                Some((_, items)) => items.push(value),
440                None => panic!("YAML list item `{value}` appeared before any group header"),
441            }
442        } else if let Some(name) = trimmed.strip_suffix(':') {
443            if line.starts_with(|c: char| c.is_whitespace()) {
444                panic!("unexpected indentation on header line: {line:?}");
445            }
446            if let Some(prev) = current.take() {
447                groups.push(prev);
448            }
449            current = Some((name.trim().to_string(), Vec::new()));
450        } else {
451            panic!("unrecognized YAML line: {line:?}");
452        }
453    }
454    if let Some(prev) = current.take() {
455        groups.push(prev);
456    }
457    groups
458}

460fn to_upper_snake(name: &str) -> String {
461    let mut out = String::with_capacity(name.len() + 4);
462    let mut prev_was_lower = false;
463    for c in name.chars() {
464        if matches!(c, '_' | '-' | ' ' | '.') {
465            if !out.ends_with('_') && !out.is_empty() {
466                out.push('_');
467            }
468            prev_was_lower = false;
469        } else if c.is_ascii_uppercase() {
470            if prev_was_lower && !out.ends_with('_') {
471                out.push('_');
472            }
473            out.push(c);
474            prev_was_lower = false;
475        } else if c.is_ascii_lowercase() {
476            out.push(c.to_ascii_uppercase());
477            prev_was_lower = true;
478        } else if c.is_ascii_digit() {
479            out.push(c);
480            prev_was_lower = false;
481        }
482    }
483    while out.ends_with('_') {
484        out.pop();
485    }
486    if out.is_empty() {
487        out.push_str("UNKNOWN");
488    }
489    if out.chars().next().is_some_and(|c| c.is_ascii_digit()) {
490        out.insert(0, '_');
491    }
492    out
493}

495fn rust_string_literal(s: &str) -> String {
496    let mut out = String::with_capacity(s.len() + 2);
497    out.push('"');
498    for c in s.chars() {
499        match c {
500            '\\' => out.push_str("\\\\"),
501            '"' => out.push_str("\\\""),
502            '\n' => out.push_str("\\n"),
503            '\r' => out.push_str("\\r"),
504            '\t' => out.push_str("\\t"),
505            c if c.is_control() => out.push_str(&format!("\\u{{{:x}}}", c as u32)),
506            c => out.push(c),
507        }
508    }
509    out.push('"');
510    out
511}

513/// Download the schemas zip and refresh every `cddl/*.cddl`,
514/// `valid_metadata_fields.yml`, and `c2pa_urn.abnf` in place.
515fn refresh_from_spec(url: &str, cddl_dir: &Path, metadata_yaml: &Path, urn_abnf: &Path) {
516    eprintln!("refresh-spec: downloading {url}");
517    let response = ureq::get(url)
518        .call()
519        .unwrap_or_else(|e| panic!("GET {url}: {e}"));
520    let mut bytes: Vec<u8> = Vec::new();
521    response
522        .into_body()
523        .into_reader()
524        .read_to_end(&mut bytes)
525        .unwrap_or_else(|e| panic!("read body from {url}: {e}"));
526
527    let cursor = std::io::Cursor::new(bytes);
528    let mut archive =
529        zip::ZipArchive::new(cursor).unwrap_or_else(|e| panic!("open zip from {url}: {e}"));
530
531    fs::create_dir_all(cddl_dir).unwrap_or_else(|e| panic!("create {}: {e}", cddl_dir.display()));
532
533    let mut cddl_count = 0usize;
534    let mut yaml_count = 0usize;
535    let mut abnf_count = 0usize;
536    for i in 0..archive.len() {
537        let mut entry = archive
538            .by_index(i)
539            .unwrap_or_else(|e| panic!("read zip entry {i}: {e}"));
540        if entry.is_dir() {
541            continue;
542        }
543        let entry_path = match entry.enclosed_name() {
544            Some(p) => p.to_path_buf(),
545            None => continue,
546        };
547        let file_name = match entry_path.file_name().and_then(|s| s.to_str()) {
548            Some(n) => n.to_string(),
549            None => continue,
550        };
551
552        let dest: Option<PathBuf> = if file_name.ends_with(".cddl") {
553            Some(cddl_dir.join(&file_name))
554        } else if file_name == "valid_metadata_fields.yml" {
555            Some(metadata_yaml.to_path_buf())
556        } else if file_name == "c2pa_urn.abnf" {
557            Some(urn_abnf.to_path_buf())
558        } else {
559            None
560        };
561        let Some(dest) = dest else { continue };
562
563        let mut buf = Vec::with_capacity(entry.size() as usize);
564        entry
565            .read_to_end(&mut buf)
566            .unwrap_or_else(|e| panic!("read {file_name} from zip: {e}"));
567        let mut f =
568            fs::File::create(&dest).unwrap_or_else(|e| panic!("create {}: {e}", dest.display()));
569        f.write_all(&buf)
570            .unwrap_or_else(|e| panic!("write {}: {e}", dest.display()));
571        if !buf.ends_with(b"\n") {
572            f.write_all(b"\n")
573                .unwrap_or_else(|e| panic!("write trailing newline to {}: {e}", dest.display()));
574        }
575        if file_name.ends_with(".cddl") {
576            cddl_count += 1;
577        } else if file_name.ends_with(".abnf") {
578            abnf_count += 1;
579        } else {
580            yaml_count += 1;
581        }
582    }
583
584    if cddl_count == 0 {
585        panic!("no .cddl entries found in archive at {url}");
586    }
587    eprintln!(
588        "refresh-spec: refreshed {cddl_count} CDDL file(s), {yaml_count} metadata YAML, {abnf_count} ABNF file(s)"
589    );
590}

592/// Pull the version segment out of a schemas-bundle URL
593/// (`.../specifications/specifications/<VERSION>/...`).
594fn extract_version_from_url(url: &str) -> Option<String> {
595    let marker = "/specifications/specifications/";
596    let start = url.find(marker)? + marker.len();
597    let rest = &url[start..];
598    let end = rest.find('/')?;
599    let candidate = &rest[..end];
600    if !candidate.is_empty()
601        && candidate.chars().all(|c| c.is_ascii_digit() || c == '.')
602    {
603        Some(candidate.to_string())
604    } else {
605        None
606    }
607}

609/// Read the spec version out of the `version = "X.Y.Z+spec-N"` line in
610/// `c2pa-spec/Cargo.toml`. Returns `None` if the file has no
611/// `+spec-<version>` build metadata.
612fn read_spec_version(cargo_toml: &Path) -> Option<String> {
613    let content = fs::read_to_string(cargo_toml).ok()?;
614    for line in content.lines() {
615        let trimmed = line.trim_start();
616        if !trimmed.starts_with("version") || !trimmed.contains("+spec-") {
617            continue;
618        }
619        let plus = trimmed.find("+spec-")? + "+spec-".len();
620        let tail = &trimmed[plus..];
621        let end = tail.find('"')?;
622        return Some(tail[..end].to_string());
623    }
624    None
625}

627/// Rewrite the `+spec-<version>` build metadata on the first
628/// `version = ...` line of `Cargo.toml` that carries it, leaving
629/// everything else untouched.
630fn write_spec_version(cargo_toml: &Path, new_version: &str) -> Result<(), String> {
631    let content = fs::read_to_string(cargo_toml)
632        .map_err(|e| format!("read {}: {e}", cargo_toml.display()))?;
633    let mut out = String::with_capacity(content.len());
634    let mut replaced = false;
635
636    for (idx, line) in content.lines().enumerate() {
637        let needs_rewrite = !replaced
638            && line.trim_start().starts_with("version")
639            && line.contains("+spec-");
640        if needs_rewrite {
641            if let (Some(plus), Some(close)) =
642                (line.find("+spec-"), line.rfind('"'))
643            {
644                let head = &line[..plus];
645                // close is the index of the closing quote; everything
646                // between plus+1 and close is the spec version segment.
647                if close > plus {
648                    let tail = &line[close..];
649                    out.push_str(head);
650                    out.push_str("+spec-");
651                    out.push_str(new_version);
652                    out.push_str(tail);
653                    out.push('\n');
654                    replaced = true;
655                    continue;
656                }
657            }
658        }
659        out.push_str(line);
660        out.push('\n');
661        let _ = idx;
662    }
663
664    if replaced {
665        fs::write(cargo_toml, out)
666            .map_err(|e| format!("write {}: {e}", cargo_toml.display()))?;
667    }
668    Ok(())
669}
